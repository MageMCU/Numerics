#include <Arduino.h>
#include "Timer.h" // Use from Joystick_Uno_L298N
#include "DeltaTime.h"
#include "Motion.h"

// GLOBAL VARIABLES
uno::Timer timer;
Numerics::DeltaTime<float> deltaTime;
Numerics::Motion<float> motion;

void setup()
{
    Serial.begin(9600);
    while (!Serial)
    {
    }

    timer = uno::Timer();
    deltaTime = Numerics::DeltaTime<float>();
    motion = Numerics::Motion<float>();
}

void loop()
{
    if (timer.isTimer(250))
    {
        // To Study Encoded Motors
        // Using Large delta time 250 ms
        // Addendum: Suggested to use gravity g = 9.8 m/s*s 
        //           y - yo = vo(t) - 0.5g(t * t);
        //           dy = - 0.5 * 9.8 * t * t;
        //   where   t = time - startTime;
        float deltaDistance = (float)random(20000, 30000);
        float dt = motion.dt(deltaTime.dt());
        float speed = motion.v(deltaDistance);
        float acceleration = motion.a(speed);

        Serial.print("dx: ");
        Serial.print(deltaDistance);
        Serial.print(" dt: ");
        Serial.print(dt);
        Serial.print(" v: ");
        Serial.print(speed);
        Serial.print(" a: ");
        Serial.println(acceleration);
    }
}
